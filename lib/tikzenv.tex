%% Create a 3th-order tensor
\tikzset{
  %% Fonction to create a cube
  cuboid/.pic={
      \tikzset{%
        every edge quotes/.append style={midway, auto},
        /cuboid/.cd,
        #1
      }
      %% Create face of tensor
      \draw[every edge/.append style={pic actions, densely dashed,
            opacity=.5}, pic actions]
      (\oox,\ooy,\ooz) coordinate (o) -- ++(-\cubescale*\cubex,\ooy,\ooz)
      coordinate (a) -- ++(\oox,-\cubescale*\cubey,\ooz) coordinate (b)
      edge coordinate [pos=1] (g) ++(\oox,\ooy,-\cubescale*\cubez)  --
      ++(\cubescale*\cubex,\ooy,\ooz) coordinate (c) -- cycle;
      \draw [every edge/.append style={pic actions, densely dashed,
            opacity=.5}, pic actions]
      (o) -- ++(\oox,\ooy,-\cubescale*\cubez) coordinate (d) --
      ++(\oox,-\cubescale*\cubey,\ooz) coordinate (e) edge (g) -- (c) -- cycle;
      \draw [every edge/.append style={pic actions, densely dashed,
            opacity=.5}, pic actions]
      (o) -- (a) -- ++(\oox,\ooy,-\cubescale*\cubez) coordinate (f) edge (g) --
      (d) -- cycle;

      % Ordre des points [a o b c][f d g e]
      \pgfextra{
        \ifthenelse{\cubenotated=3}{
          %% Legend of tensor with edge
          %% A vérifier pourquoi ca bugg chez Théo!!
          \path [every edge/.append style={pic actions, |-|}]
          (b) +(\oox,-0.2pt+\ooy,\ooz) coordinate (b1) edge [below,"\lx"'] (b1
          -| c)
          (b) +(-.2pt+\oox,\ooy,\ooz) coordinate (b2) edge ["\ly"] (b2 |- a)
          (c) +(0.1pt+\oox,-0.15pt+\ooy,\ooz) coordinate (c2) edge ["\lz"']
          ([xshift=3.5pt,yshift=-3.5pt]e);
          
        }{
          \ifthenelse{\cubenotated=2}{
            %% Legend of tensor
            \coordinate (c2) at ($(c)+(0.1+\oox,-0.1+\ooy,\ooz)$);
            \coordinate (c3) at ($(c2)+(0.1,0,-\cubescale*\cubez*0.5)$);
            \node[xshift=\cubescale*\cubex*0.5 cm, yshift = -0.2cm] at (b)
            {\lx};
            \node[xshift=-0.2cm, yshift=\cubey*\cubescale*0.5 cm] at (b) {\ly};
            \node[] at (c3) {\lz};
          }{}
        }
        %% Name of tensor
        \node[anchor=center, scale=\cubex/10] at
        (0.1+\oox,\ooy,\cubez*0.1+\ooz) {\g};
      }
    },
  %% All parameters
  /cuboid/.search also={/tikz},
  /cuboid/.cd,
  width/.store in=\cubex,
  height/.store in=\cubey,
  depth/.store in=\cubez,
  units/.store in=\cubeunits,
  scale/.store in=\cubescale,
  nameG/.store in=\g,
  originX/.store in=\oox,
  originY/.store in=\ooy,
  originZ/.store in=\ooz,
  legendX/.store in=\lx,
  legendY/.store in=\ly,
  legendZ/.store in=\lz,
  notated/.store in=\cubenotated,
  %% Initialisation of parameters
  width=10,
  height=10,
  depth=10,
  units=cm,
  scale=.1,
  originX=0,
  originY=0,
  originZ=0,
  legendX=\footnotesize{$m_1$},
  legendY=\footnotesize{$m_2$},
  legendZ=\footnotesize{$m_3$},
  notated=1,
  nameG=0,
}
